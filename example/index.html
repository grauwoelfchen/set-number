SetNumber Example

Example

Usage

Load jQuery and SetNumber (js, css)

          ;; e.g. CDN
          <script type="text/javascript" src="http://ajax.googleapis.com/ajax/libs/jquery/1.10.2/jquery.min.js"></script>
          <script type="text/javascript" src="http://cdnjs.cloudflare.com/ajax/libs/autosize.js/1.18.1/jquery.autosize.min.js"></script>

          ;; SetNumber
          <script type="text/javascript" src="../path/to/set-number.min.js"></script>
          <link rel="stylesheet" href="../path/to/set-number.css" media="screen" />

Put textarea

          <body>
            <textarea id="editor"></textarea>
          </body>

Apply setNumber

          <script>
            var ready = function() {
              // set-number
              $("#editor").setNumber({
                activeLine: 9
              });
              // jquery-autosize (optional)
              $("#editor").autosize({
                callback: function(textarea) {
                  $(textarea).scroll();
                }
              });
            };
            $(document).ready(ready);
          </script>
